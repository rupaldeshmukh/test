function modifyString(text) {
  // Split the string into words using a space as the delimiter.
  const words = text.split(" ");

  // Initialize an empty array to store the modified words.
  const modifiedWords = [];

  // Iterate through each word in the original string.
  for (let i = 0; i < words.length; i++) {
    const word = words[i];

    // Add "and domain is finance" before the current "or" if not last word.
    if (word === "or" && i < words.length - 1) {
      modifiedWords.push("and domain is finance");
    }

    // Add the current word to the modified words array.
    modifiedWords.push(word);
  }

  // Add "and domain is finance" at the end of the string.
  modifiedWords.push("and domain is finance");

  // Join the modified words back into a string with spaces in between.
  return modifiedWords.join(" ");
}

// Example usage
const text = "This string contains the word or multiple times.";
const modifiedText = modifyString(text);
console.log(modifiedText); // Output: This string contains the word and domain is finance or multiple times and domain is finance.
