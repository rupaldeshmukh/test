import React, { useState } from 'react';

const products = [
  { id: 1, name: 'Product 1', certified: 'silver' },
  { id: 2, name: 'Product 2', certified: 'gold' },
  { id: 3, name: 'Product 3', certified: 'silver' },
  { id: 4, name: 'Product 4', certified: 'silver certified' },
  // ... more products
];

function App() {
  const [selectedFilters, setSelectedFilters] = useState([]);

  const handleCheckboxChange = (event) => {
    const isChecked = event.target.checked;
    const value = event.target.value;

    if (isChecked) {
      setSelectedFilters([...selectedFilters, value]);
    } else {
      setSelectedFilters(selectedFilters.filter((filter) => filter !== value));
    }
  };

  const filteredProducts = products.filter((product) => {
    if (selectedFilters.length === 0) {
      return true; // Show all products if no filters selected
    }

    return selectedFilters.some((filter) => {
      if (filter === 'silver certified') {
        return product.certified === 'silver'; // Handle special "silver certified" case
      } else {
        return product.certified.toLowerCase() === filter.toLowerCase(); // Case-insensitive filter comparison
      }
    });
  });

  return (
    <div>
      <h1>Filter Products</h1>
      <div>
        <label>
          <input
            type="checkbox"
            value="silver"
            checked={selectedFilters.includes('silver')}
            onChange={handleCheckboxChange}
          />
          Silver
        </label>
        <label>
          <input
            type="checkbox"
            value="gold"
            checked={selectedFilters.includes('gold')}
            onChange={handleCheckboxChange}
          />
          Gold
        </label>
        <label>
          <input
            type="checkbox"
            value="silver certified"
            checked={selectedFilters.includes('silver certified')}
            onChange={handleCheckboxChange}
          />
          Silver Certified
        </label>
      </div>
      <ul>
        {filteredProducts.map((product) => (
          <li key={product.id}>{product.name}</li>
        ))}
      </ul>
    </div>
  );
}

export default App;
